import random import numpy as np from PIL import Image, ImageDraw def generate_dungeon(width=50, height=40, num_tunnels=150, max_tunnel_len=18, num_rooms=10): """Random walk + small rooms for a nice dungeon feel.""" grid = np.ones((height, width), dtype=int) # 1 = wall, 0 = floor # Start point x = random.randint(5, width - 6) y = random.randint(5, height - 6) grid[y, x] = 0 directions = [(-1, 0), (1, 0), (0, -1), (0, 1)] # Carve tunnels for _ in range(num_tunnels): dx, dy = random.choice(directions) length = random.randint(4, max_tunnel_len) for _ in range(length): nx, ny = x + dx, y + dy if not (2 <= nx < width - 2 and 2 <= ny < height - 2): break x, y = nx, ny grid[y, x] = 0 # Add a few small rectangular rooms for _ in range(num_rooms): room_w = random.randint(3, 7) room_h = random.randint(3, 6) rx = random.randint(2, width - room_w - 3) ry = random.randint(2, height - room_h - 3) for i in range(room_h): for j in range(room_w): if 0 <= ry + i < height and 0 <= rx + j < width: grid[ry + i, rx + j] = 0 return grid # === Generate the map === grid = generate_dungeon() # Quarter-inch settings (300 DPI) CELL_INCHES = 0.25 DPI = 300 CELL_SIZE = int(CELL_INCHES * DPI) # 75 pixels img_width = grid.shape[1] * CELL_SIZE img_height = grid.shape[0] * CELL_SIZE img = Image.new('RGB', (img_width, img_height), color='#f8f6f0') # warm paper tone draw = ImageDraw.Draw(img) # Fill walls (dark) wall_color = '#1f1f1f' for y in range(grid.shape[0]): for x in range(grid.shape[1]): if grid[y, x] == 1: draw.rectangle( [x * CELL_SIZE, y * CELL_SIZE, (x + 1) * CELL_SIZE, (y + 1) * CELL_SIZE], fill=wall_color ) # Thin graph-paper grid lines (light gray) grid_color = '#d0d0d0' line_width = 2 for x in range(grid.shape[1] + 1): draw.line([(x * CELL_SIZE, 0), (x * CELL_SIZE, img_height)], fill=grid_color, width=line_width) for y in range(grid.shape[0] + 1): draw.line([(0, y * CELL_SIZE), (img_width, y * CELL_SIZE)], fill=grid_color, width=line_width) # Optional thicker outer border draw.rectangle([0, 0, img_width - 1, img_height - 1], outline='#444444', width=8) # Save with correct DPI img.save('quarter_inch_dungeon_grid.png', dpi=(DPI, DPI)) print("✅ Saved as 'quarter_inch_dungeon_grid.png'") print(f" Grid size: {grid.shape[1]} × {grid.shape[0]} cells") print(f" Physical print size: {grid.shape[1] * CELL_INCHES:.1f}″ × {grid.shape[0] * CELL_INCHES:.1f}″") print(" Print at 100% scale for perfect ¼-inch squares!")
